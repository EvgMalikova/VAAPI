29 __device__ bool refract(const optix::float3& v, const optix::float3& n, float ni_over_nt, optix::float3& refracted) {
30  optix::float3 uv = optix::normalize(v);
31  float dt = optix::dot(uv, n);
32  float discriminant = 1.0f - ni_over_nt * ni_over_nt*(1 - dt * dt);
33  if (discriminant > 0) {
34  refracted = ni_over_nt * (uv - n * dt) - n * sqrtf(discriminant);
35  return true;
36  }
37  else
38  return false;
39 }
