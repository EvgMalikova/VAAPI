36 // Tiny Encryption Algorithm (TEA) to calculate a the seed per launch index and iteration.
37 template<unsigned int N>
38 RT_FUNCTION unsigned int tea(const unsigned int val0, const unsigned int val1)
39 {
40  unsigned int v0 = val0;
41  unsigned int v1 = val1;
42  unsigned int s0 = 0;
43 
44  for (unsigned int n = 0; n < N; ++n)
45  {
46  s0 += 0x9e3779b9;
47  v0 += ((v1 << 4) + 0xA341316C) ^ (v1 + s0) ^ ((v1 >> 5) + 0xC8013EA4);
48  v1 += ((v0 << 4) + 0xAD90777D) ^ (v0 + s0) ^ ((v0 >> 5) + 0x7E95761E);
49  }
50  return v0;
51 }
